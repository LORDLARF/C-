            function saveState() {
                localStorage.setItem('appState', JSON.stringify(State));
            }

            function loadState() {
                const saved = localStorage.getItem('appState');
                if (saved) {
                    try {
                        const parsed = JSON.parse(saved);
                        State.currentUser = parsed.currentUser || null;
                        State.currentPage = parsed.currentPage || 'landing';
                        State.theme = parsed.theme || 'light';
                        State.selectedCourse = parsed.selectedCourse || null;
                        State.adminSubPage = parsed.adminSubPage || 'dashboard';
                        State.enrolledCourses = parsed.enrolledCourses || [];
                        State.currentLesson = parsed.currentLesson || null;
                        State.welcomeShown = parsed.welcomeShown || false;
                    } catch (e) {
                        console.error('Failed to load state', e);
                    }
                }
            }

            function renderApp() {
                renderNavbar();
                renderPage();
                saveState();
            }

            function navigate(page, data = null) {
                State.currentPage = page;
                if (data) {
                    State.selectedCourse = data;
                } else if (page !== 'courseDetail' && page !== 'myCourse') {
                    State.selectedCourse = null;
                }
                if (page !== 'myCourse') {
                    State.currentLesson = null;
                }
                renderApp();
            }

            function loginUser(username, password) {
                const found = usersDB.find(u => u.username === username && u.password === password);
                if (found) {
                    State.currentUser = { username: found.username, role: found.role };
                    State.currentPage = (found.role === 'admin') ? 'admin' : 'dashboard';
                    renderApp();
                    return true;
                }
                return false;
            }

            function registerUser(username, password) {
                if (!username.trim() || !password.trim()) {
                    return { success: false, message: 'Заполните все поля' };
                }
                if (usersDB.some(u => u.username === username)) {
                    return { success: false, message: 'Пользователь с таким именем уже существует' };
                }
                usersDB.push({ username, password, role: 'user' });
                return { success: true };
            }

            function logout() {
                State.currentUser = null;
                State.currentPage = 'landing';
                State.enrolledCourses = [];
                State.selectedCourse = null;
                State.currentLesson = null;
                renderApp();
            }

            function enrollCourse(courseId) {
                if (!State.enrolledCourses.includes(courseId)) {
                    State.enrolledCourses.push(courseId);
                }
                const course = coursesData.find(c => c.id === courseId);
                navigate('myCourse', course);
            }

            function renderPage() {
                const root = document.getElementById('appRoot');
                if (!root) return;
                const page = State.currentPage;
                const user = State.currentUser;

                if (!user && (page === 'dashboard' || page === 'admin' || page === 'myCourse')) {
                    navigate('login');
                    return;
                }

                switch (page) {
                    case 'landing': renderLanding(root); break;
                    case 'dashboard': renderDashboard(root); break;
                    case 'courses': renderCourses(root); break;
                    case 'courseDetail': renderCourseDetail(root); break;
                    case 'myCourse': renderMyCourse(root); break;
                    case 'about': renderAbout(root); break;
                    case 'admin': renderAdmin(root); break;
                    case 'login': renderLogin(root); break;
                    case 'register': renderRegister(root); break;
                    default: renderLanding(root);
                }
            }

                document.getElementById('loginSubmit').addEventListener('click', () => {
                    const u = document.getElementById('loginUsername').value.trim();
                    const p = document.getElementById('loginPassword').value;
                    const errorEl = document.getElementById('loginError');

                    if (!u || !p) {
                        errorEl.textContent = 'Введите логин и пароль';
                        errorEl.classList.add('show');
                        return;
                    }

                    const success = loginUser(u, p);
                    if (!success) {
                        errorEl.textContent = 'Неверный логин или пароль. Проверьте данные или зарегистрируйтесь.';
                        errorEl.classList.add('show');
                    } else {
                        errorEl.classList.remove('show');
                    }
                });
                document.getElementById('toRegister').addEventListener('click', (e) => {
                    e.preventDefault();
                    navigate('register');
                });
            }

            function renderRegister(container) {
                container.innerHTML = `
                    <div class="auth-form">
                        <h2>Регистрация</h2>

                        <p style="margin-top:1rem;">Уже есть аккаунт? <a id="toLogin" style="cursor:pointer; color:var(--primary);">Войти</a></p>
                    </div>
                `;
                document.getElementById('regSubmit').addEventListener('click', () => {
                    const u = document.getElementById('regUsername').value.trim();
                    const p = document.getElementById('regPassword').value;
                    const errorEl = document.getElementById('regError');

                    const result = registerUser(u, p);
                    if (!result.success) {
                        errorEl.textContent = result.message;
                        errorEl.classList.add('show');
                    } else {
                        errorEl.classList.remove('show');
                        State.currentUser = { username: u, role: 'user' };
                        State.currentPage = 'dashboard';
                        renderApp();
                    }
                });
                document.getElementById('toLogin').addEventListener('click', (e) => {
                    e.preventDefault();
                    navigate('login');
                });
            }

            function showWelcomeScreen() {
                const welcomeScreen = document.getElementById('welcomeScreen');
                if (!welcomeScreen) return;

                if (State.welcomeShown) {
                    welcomeScreen.classList.add('fade-out');
                    setTimeout(() => {
                        welcomeScreen.style.display = 'none';
                    }, 500);
                    return;
                }

                welcomeScreen.style.display = 'flex';
                document.getElementById('startExploring')?.addEventListener('click', () => {
                    State.welcomeShown = true;
                    saveState();
                    welcomeScreen.classList.add('fade-out');
                    setTimeout(() => {
                        welcomeScreen.style.display = 'none';
                        if (State.currentUser && State.currentPage === 'landing') {
                            State.currentPage = State.currentUser.role === 'admin' ? 'admin' : 'dashboard';
                        }
                        renderApp();
                    }, 500);
                });
            }

            loadState();
            showWelcomeScreen();
            if (State.welcomeShown) {
                document.getElementById('welcomeScreen').style.display = 'none';
                if (State.currentUser && State.currentPage === 'landing') {
                    State.currentPage = State.currentUser.role === 'admin' ? 'admin' : 'dashboard';
                }
                renderApp();
            }
        })();
